function( FcnBuildExample paramSources paramExecutableName )

    add_executable( ${paramExecutableName}
                    ${paramSources} )

    target_link_libraries( ${paramExecutableName}
                           tinyrenderer_cpp_lib
                           tinymath_cpp_lib )

endfunction()
