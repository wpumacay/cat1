set( TINYRENDERER_IS_MASTER_PROJECT OFF )

    message( "TINYRENDERER> Configuring as MASTER project" )
    set( TINYRENDERER_IS_MASTER_PROJECT ON )
else()
    message( "TINYRENDERER> Configuring as CHILD project" )
